# source TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_256_1518B/TEST_333_459_518_520B/script_10G.txt
# script for testing LMAC_STREAM :  generating tx packets
# gen_en_wr testbench signals generates tx packets which is send from AXIS bridge to LMAC



vsim work.axis_lmac_tb            
                                   
do C:/LMAC3_INFO/waveforms/tx_10G

mem load -i C:/LMAC3_INFO/TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_256_1518B/TEST_333_459_518_520B/memory_wr_data333_459_518_520B.txt -format hex /AXIS_LMAC_TB/axi_stream_master/memory_wr_module/memory_wr_data
mem load -i C:/LMAC3_INFO/TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_256_1518B/TEST_333_459_518_520B/memory_wr_ctrl333_459_518_520B.txt -format hex /AXIS_LMAC_TB/axi_stream_master/memory_wr_module/memory_wr_ctrl

mem load -i C:/LMAC3_INFO/PHY_EMULATOR/rx_pkt_gen_data.mem -format hex /AXIS_LMAC_TB/phy_emulator_100G/rx_pkt_gen_100G/rx_pkt_gen2kx256_inst/dram_data/mem 
mem load -i C:/LMAC3_INFO/PHY_EMULATOR/rx_pkt_gen_ctrl.mem -format hex /AXIS_LMAC_TB/phy_emulator_100G/rx_pkt_gen_100G/rx_pkt_gen2kx32_inst/dram_ctrl/mem  

######Clock assignment for speed - 10G
force -freeze sim:/AXIS_LMAC_TB/clk 0 0, 1 {1600 ps} -r {3200 ps}

force -freeze sim:/AXIS_LMAC_TB/tx_mac_aclk 0 0, 1 {1600 ps} -r {3200 ps}
force -freeze sim:/AXIS_LMAC_TB/rx_mac_aclk 0 0, 1 {1600 ps} -r {3200 ps}

force -freeze sim:/AXIS_LMAC_TB/xA_clk 0 0, 1 {3200 ps} -r {6400 ps}                                                                                                                                                           
                                                                                                                                                           
force -freeze sim:/AXIS_LMAC_TB/fmac_speed 3'b000 0                                                                                                        
force -freeze sim:/AXIS_LMAC_TB/phy_emulator_100G/rx_pkt_gen_start_addr 64'd00 0                                                                           
                                                                                                                                                           
                                                                                                                                                           
                                                                                                                                                           
# generating first packet     - TX packet 1                                                                                                                
                                                                                                                                                           
run 300ns                                                                                                                                                  
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0                                                                                                              
run 3.2ns                                                                                                                                                    
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0                                                                                                              
                                                                                                                                                           
# Generate the TX packet	-- TX packet 2                                                                                                                    
run 200ns                                                                                                                                                  
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0                                                                                                              
run 3.2ns                                                                                                                                                    
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0                                                                                                              
                                                                                                                                                           
# Generate the TX packet	-- TX packet 3                                                                                                                    
run 200ns                                                                                                                                                  
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0                                                                                                              
run 3.2ns                                                                                                                                                    
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0                                                                                                              
                                                                                                                                                           
run 200ns                                                                                                                                                  
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0                                                                                                              
run 3.2ns                                                                                                                                                    
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0                                                                                                              
                                                                                                                                                           
run 3us
